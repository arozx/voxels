117 void TerrainSystem::GenerateMesh() {
118 std::vector<float> vertices;
119 std::vector<uint32_t> indices;
120 uint32_t currentIndex = 0;
121
122 // Increase map size for better visibility
123 int mapSize = (m_ChunkRange * 2 + 1) * 16;
124 auto heightmap = m_NoiseGen.getHeightmap(mapSize, mapSize, m_NoiseScale);
125
126 // Generate quad mesh with proper UVs
127 for (int z = 0; z < mapSize - 1; z++) {
128 for (int x = 0; x < mapSize - 1; x++) {
129 float height00 = m_BaseHeight + heightmap[z * mapSize + x] * m_HeightScale;
130 float height10 = m_BaseHeight + heightmap[z * mapSize + (x + 1)] * m_HeightScale;
131 float height01 = m_BaseHeight + heightmap[(z + 1) * mapSize + x] * m_HeightScale;
132 float height11 = m_BaseHeight + heightmap[(z + 1) * mapSize + (x + 1)] * m_HeightScale;
133
134 // Add vertices for this quad
135 vertices.insert(vertices.end(), {
136 static_cast<float>(x), height00, static_cast<float>(z), 0.0f, 0.0f,
137 static_cast<float>(x + 1), height10, static_cast<float>(z), 1.0f, 0.0f,
138 static_cast<float>(x + 1), height11, static_cast<float>(z + 1), 1.0f, 1.0f,
139 static_cast<float>(x), height01, static_cast<float>(z + 1), 0.0f, 1.0f
140 });
141
142 // Add indices for the quad (two triangles)
143 indices.insert(indices.end(), {
144 currentIndex, currentIndex + 1, currentIndex + 2,
145 currentIndex, currentIndex + 2, currentIndex + 3
146 });
147
148 currentIndex += 4;
149 }
150 }
151
152 // Create vertex array with mesh data
153 m_TerrainVA.reset(VertexArray::Create());
154
155 if (!vertices.empty()) {
156 std::shared_ptr<VertexBuffer> vertexBuffer(
157 VertexBuffer::Create(vertices.data(),
158 vertices.size() * sizeof(float)));
159
160 BufferLayout layout = {
161 { ShaderDataType::Float3, "aPosition" },
162 { ShaderDataType::Float2, "aTexCoord" }
163 };
164
165 vertexBuffer->SetLayout(layout);
166 m_TerrainVA->AddVertexBuffer(vertexBuffer);
167
168 std::shared_ptr<IndexBuffer> indexBuffer(
169 IndexBuffer::Create(indices.data(), indices.size()));
170 m_TerrainVA->SetIndexBuffer(indexBuffer);
171 }
172
173 // Debug output
174 LOG_TRACE_CONCAT("Generated terrain mesh with ", vertices.size() / 5, " vertices and ",
175 indices.size(), " indicies.");
176 }
