176 }
177
178 // Add frame profiling input control
179 static int framesToProfile = 60;
180 ImGui::SetNextItemWidth(100);
181 ImGui::InputInt("Frames to Profile", &framesToProfile);
182 framesToProfile = std::max(1, std::min(framesToProfile, 1000)); // Clamp between 1-1000
183
184 if (!Profiler::Get().IsProfilingFrames() && ImGui::Button("Profile Frames")) {
185 Profiler::Get().BeginSession("Frame Profile");
186 Profiler::Get().ProfileFrames(framesToProfile);
187 }
188
189 // Show profiling progress if active
190 if (Profiler::Get().IsProfilingFrames()) {
191 ImGui::SameLine();
192 ImGui::Text("Profiling Frame: %d/%d",
193 Profiler::Get().GetCurrentProfiledFrame(),
194 framesToProfile);
195 }
196
197 ImGui::Separator();
198
199 for (const auto& [name, timings] : Profiler::Get().GetProfiles()) {
200 if (timings.empty()) continue;
201
202 float total = 0.0f;
203 for (float time : timings) {
204 total += time;
205 }
206
207 float avg = total / static_cast<float>(timings.size());
208 float min = *std::min_element(timings.begin(), timings.end());
209 float max = *std::max_element(timings.begin(), timings.end());
210
211 ImGui::Text("%s:", name.c_str());
212 ImGui::Text(" Avg: %.3f ms", avg);
213 ImGui::Text(" Min: %.3f ms", min);
214 ImGui::Text(" Max: %.3f ms", max);
215 ImGui::Text(" Calls: %zu", timings.size());
216 ImGui::Separator();
