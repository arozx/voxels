79void Renderer::Flush() {
80 // Assert we're not recursively processing frames
81 if (m_ProcessingFrame.exchange(true)) {
82 ASSERT(false && "Already processing a frame");
83 return; // Already processing a frame
84 }
85
86 std::vector<PreprocessedRenderCommand> commands;
87 {
88 std::lock_guard<std::mutex> lock(m_QueueMutex);
89 while (!m_CommandQueue.empty()) {
90 auto cmd = m_CommandQueue.front();
91 commands.push_back(
92 {cmd.vertexArray, cmd.material, cmd.primitiveType, cmd.transformMatrix});
93 m_CommandQueue.pop();
94 }
95 }
96
97 static bool taskSystemInitialized = false;
98 if (!taskSystemInitialized) {
99 TaskSystem::Get().Initialize();
100 taskSystemInitialized = true;
101 }
102
103 // Process commands in parallel using TaskSystem
104 std::vector<std::future<void>> futures;
105 const size_t batchSize = 64; // Number of commands to process per task
106
107 for (size_t i = 0; i < commands.size(); i += batchSize) {
108 size_t end = std::min(i + batchSize, commands.size());
109 futures.push_back(TaskSystem::Get().EnqueueTask([this, &commands, i, end]() {
110 for (size_t j = i; j < end; j++) {
111 auto& cmd = commands[j];
112 // Do CPU-intensive work here (transformations, frustum culling, etc.)
113 }
114 }));
115 }
116
117 for (auto& future : futures) {
118 future.wait();
119 }
120
121 // Execute render commands on main thread
122 for (const auto& command : commands) {
123 // Assert required components are valid before rendering
124 ASSERT(command.vertexArray->GetIndexBuffer() != nullptr && "Null index buffer");
125 ASSERT(command.vertexArray->GetIndexBuffer()->GetCount() > 0 && "Empty index buffer");
126
127 command.material->Bind();
128 auto shader = command.material->GetShader();
129 ASSERT(shader != nullptr && "Null shader in material");
130
131 if (m_CameraType == CameraType::Orthographic) {
132 shader->SetMat4("u_ViewProjection", m_Camera->GetViewProjectionMatrix());
133 } else {
134 shader->SetMat4("u_ViewProjection", m_PerspectiveCamera->GetViewProjectionMatrix());
135 }
136 shader->SetMat4("u_Model", command.modelMatrix);
137
138 command.vertexArray->Bind();
139 glDrawElements(command.primitiveType, command.vertexArray->GetIndexBuffer()->GetCount(),
140 GL_UNSIGNED_INT, nullptr);
141
142 command.vertexArray->Unbind();
143 command.material->Unbind();
144 }
