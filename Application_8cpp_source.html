194 void Application::ProcessEvents() {
195 static float eventTimer = 0.0f;
196 static const float fixedDeltaTime = 1.0f / 60.0f;
197 eventTimer += fixedDeltaTime;
198
199 // Process events at fixed 60Hz rate
200 if (eventTimer >= EVENT_PROCESS_INTERVAL) {
201 eventTimer = 0.0f;
202
203 EventQueue::Get().ProcessEvents([this](std::shared_ptr<Event> event) {
204 EventDispatcher dispatcher(*event.get());
205
206 dispatcher.Dispatch<WindowCloseEvent>([this](const WindowCloseEvent& e [[maybe_unused]]) {
207 LOG_INFO("Window Close Event received");
208 m_Running = false;
209 return true;
210 });
211
212 if (!event->IsHandled()) {
213 m_InputSystem->OnEvent(*event);
214 }
215 return true;
216 });
217
218 // Replace GLFW key checking with input system
219 for (int i = 0; i <= 9; i++) {
220 if (m_InputSystem->IsKeyPressed(KeyCode::NUM_0 + i)) {
221 auto event = std::make_shared<KeyPressedEvent>(KeyCode::NUM_0 + i);
222 EventQueue::Get().PushEvent(event);
223 break;
224 }
225 }
226 }

256 void Application::InitWindow(const char* title, int width, int height) {
257 LOG_TRACE_CONCAT("Creating window: ", title, ", Resolution: ", width, "x", height);
258 WindowProps props(title, width, height);
259 m_Window = std::unique_ptr<Window>(Window::Create(props));
260
261 m_Window->SetEventCallback([this](const Event& e) {
262 std::shared_ptr<Event> eventPtr = nullptr;
263
264 if (e.GetEventType() == EventType::WindowResize) {
265 const auto* resizeEvent = static_cast<const WindowResizeEvent*>(&e);
266 SetViewport(0, 0, resizeEvent->GetWidth(), resizeEvent->GetHeight());
267 }
268
269 // Clone the event based on its type
270 if (e.GetEventType() == EventType::KeyPressed) {
271 const auto* keyEvent = static_cast<const KeyPressedEvent*>(&e);
272 eventPtr = std::make_shared<KeyPressedEvent>(keyEvent->GetKeyCode(), keyEvent->IsRepeat());
273 }
274 else if (e.GetEventType() == EventType::MouseMoved) {
275 const auto* mouseEvent = static_cast<const MouseMovedEvent*>(&e);
276 eventPtr = std::make_shared<MouseMovedEvent>(mouseEvent->GetX(), mouseEvent->GetY());
277 }
278
279 if (eventPtr) {
280 EventQueue::Get().PushEvent(eventPtr);
281 }
282 return false;
283 });
284 m_Window->SetContext();
285 }

373 bool Application::HandleKeyToggle(int key, float currentTime) {
374 auto& state = m_KeyToggles[key];
375 bool currentState = m_InputSystem->IsKeyPressed(key);
376
377 // Track when key is first pressed
378 if (currentState && !state.previousState) {
379 state.pressStartTime = currentTime;
380 }
381
382 // Check for toggle condition on release
383 if (state.previousState && !currentState) {
384 bool shouldToggle = (currentTime - state.pressStartTime) < MAX_TOGGLE_HOLD_TIME;
385 state.previousState = currentState;
386 if (shouldToggle) {
387 state.currentValue = !state.currentValue;
388 return true;
389 }
390 }
391
392 state.previousState = currentState;
393 return false;
394 }

416 void Application::ConfigureForRenderType() {
417 auto activeScene = SceneManager::Get().GetActiveScene();
418 if (!activeScene) return;
419
420 switch (m_RenderType) {
421 case RenderType::Render2D:
422 if (m_TerrainSystem) {
423 m_TerrainSystem->Shutdown();

426 activeScene->SetCameraType(CameraType::Orthographic);
427 break;
428

432 if (m_TerrainSystem) {
433 m_TerrainSystem->Initialize(GetRenderer());
