208 void Application::ProcessEvents() {
209 static float eventTimer = 0.0f;
210 static const float fixedDeltaTime = 1.0f / 60.0f;
211 eventTimer += fixedDeltaTime;
212
213 // Process events at fixed 60Hz rate
214 if (eventTimer >= EVENT_PROCESS_INTERVAL) {
215 eventTimer = 0.0f;
216
217 EventQueue::Get().ProcessEvents([this](std::shared_ptr<Event> event) {
218 EventDispatcher dispatcher(*event.get());
219
220 dispatcher.Dispatch<WindowCloseEvent>([this](const WindowCloseEvent& e [[maybe_unused]]) {
221 LOG_INFO("Window Close Event received");
222 m_Running = false;
223 return true;
224 });
225
226 if (!event->IsHandled()) {
227 m_InputSystem->OnEvent(*event);
228 }
229 return true;
230 });
231
232 // Replace GLFW key checking with input system
233 for (int i = 0; i <= 9; i++) {
234 if (m_InputSystem->IsKeyPressed(KeyCode::NUM_0 + i)) {
235 auto event = std::make_shared<KeyPressedEvent>(KeyCode::NUM_0 + i);
236 EventQueue::Get().PushEvent(event);
237 break;
238 }
239 }
240 }

270 void Application::InitWindow(const char* title, int width, int height) {
271 LOG_TRACE_CONCAT("Creating window: ", title, ", Resolution: ", width, "x", height);
272 WindowProps props(title, width, height);
273 m_Window = std::unique_ptr<Window>(Window::Create(props));
274
275 m_Window->SetEventCallback([this](const Event& e) {
276 std::shared_ptr<Event> eventPtr = nullptr;
277
278 if (e.GetEventType() == EventType::WindowResize) {
279 const auto* resizeEvent = static_cast<const WindowResizeEvent*>(&e);
280 SetViewport(0, 0, resizeEvent->GetWidth(), resizeEvent->GetHeight());
281 }
282
283 // Clone the event based on its type
284 if (e.GetEventType() == EventType::KeyPressed) {
285 const auto* keyEvent = static_cast<const KeyPressedEvent*>(&e);
286 eventPtr = std::make_shared<KeyPressedEvent>(keyEvent->GetKeyCode(), keyEvent->IsRepeat());
287 }
288 else if (e.GetEventType() == EventType::MouseMoved) {
289 const auto* mouseEvent = static_cast<const MouseMovedEvent*>(&e);
290 eventPtr = std::make_shared<MouseMovedEvent>(mouseEvent->GetX(), mouseEvent->GetY());
291 }
292
293 if (eventPtr) {
294 EventQueue::Get().PushEvent(eventPtr);
295 }
296 return false;
297 });
298 m_Window->SetContext();
299 }

387 bool Application::HandleKeyToggle(int key, float currentTime) {
388 auto& state = m_KeyToggles[key];
389 bool currentState = m_InputSystem->IsKeyPressed(key);
390
391 // Track when key is first pressed
392 if (currentState && !state.previousState) {
393 state.pressStartTime = currentTime;
394 }
395
396 // Check for toggle condition on release
397 if (state.previousState && !currentState) {
398 bool shouldToggle = (currentTime - state.pressStartTime) < MAX_TOGGLE_HOLD_TIME;
399 state.previousState = currentState;
400 if (shouldToggle) {
401 state.currentValue = !state.currentValue;
402 return true;
403 }
404 }
405
406 state.previousState = currentState;
407 return false;
408 }

430 void Application::ConfigureForRenderType() {
431 auto activeScene = SceneManager::Get().GetActiveScene();
432 if (!activeScene) return;
433
434 switch (m_RenderType) {
435 case RenderType::Render2D:
436 if (m_TerrainSystem) {
437 m_TerrainSystem->Shutdown();
438 m_TerrainSystem = nullptr;

440 activeScene->SetCameraType(CameraType::Orthographic);
441 break;
442
443 case RenderType::Render3D:
444 if (!m_TerrainSystem && activeScene) {
445 m_TerrainSystem = activeScene->GetTerrainSystem();
446 if (m_TerrainSystem) {
447 m_TerrainSystem->Initialize(GetRenderer());

449 }
450 activeScene->SetCameraType(CameraType::Perspective);
451 break;
