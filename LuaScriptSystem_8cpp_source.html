542bool LuaScriptSystem::ExecuteScript(const std::string& script) {
543 sol::load_result loadResult = m_LuaState->load(script);
544 if (!loadResult.valid()) {
545 sol::error err = loadResult;
546 LOG_ERROR_CONCAT("Failed to load script: ", err.what());
547 return false;
548 }
549
550 sol::protected_function_result result = loadResult();
551 if (!result.valid()) {
552 sol::error err = result;
553 LOG_ERROR_CONCAT("Failed to execute script: ", err.what());
554 return false;
555 }
556
557 return true;
558}

581bool LuaScriptSystem::ExecuteFile(const std::string& originalPath) {
582 std::vector<std::string> searchPaths = {
583 originalPath, "../" + originalPath, "../../" + originalPath,
584 "../sandbox/assets/scripts/" + originalPath.substr(originalPath.find_last_of("/\\") + 1),
585 "sandbox/assets/scripts/" + originalPath.substr(originalPath.find_last_of("/\\") + 1)};
586
587 std::string validPath;
588 bool found = false;
589
590 for (const auto& path : searchPaths) {
591 if (FileSystem::Exists(path)) {
592 validPath = path;
593 found = true;
594 break;
595 }
596 }
597
598 if (!found) {
599 LOG_ERROR("Script not found: ", originalPath);
600 return false;
601 }
602
603 sol::load_result loadResult = m_LuaState->load_file(validPath);
604 if (!loadResult.valid() || !loadResult().valid()) {
605 LOG_ERROR("Failed to execute script: ", validPath);
606 return false;
607 }
608
609 return true;
610}
