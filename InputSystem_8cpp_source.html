95 void InputSystem::HandleCameraMovement(float deltaTime) {
96 if (!m_CameraEnabled || m_MovementLocked) return;
97
98 auto& camera = m_Renderer.GetCamera();
99 if (!camera) return;
100
101 float speed = m_MovementSpeed;
102 HandleSpeedModifiers(speed);
103
104 if (m_Renderer.GetCameraType() == Renderer::CameraType::Perspective) {
105 auto& perspCamera = m_Renderer.GetPerspectiveCamera();
106 if (perspCamera) {
107 glm::vec3 moveDir(0.0f);
108
109 // Get camera orientation vectors
110 glm::vec3 forward = glm::normalize(glm::vec3(perspCamera->GetFront().x, 0.0f, perspCamera->GetFront().z));
111 glm::vec3 right = glm::normalize(glm::cross(forward, glm::vec3(0.0f, 1.0f, 0.0f)));
112
113 // Move relative to camera orientation
114 if (IsKeyPressed(GLFW_KEY_W)) moveDir += forward;
115 if (IsKeyPressed(GLFW_KEY_S)) moveDir -= forward;
116 if (IsKeyPressed(GLFW_KEY_A)) moveDir -= right;
117 if (IsKeyPressed(GLFW_KEY_D)) moveDir += right;
118 if (IsKeyPressed(GLFW_KEY_SPACE)) moveDir.y += 1.0f;
119 if (IsKeyPressed(GLFW_KEY_LEFT_SHIFT)) moveDir.y -= 1.0f;
120
121 if (glm::length(moveDir) > 0.0f) {
122 moveDir = glm::normalize(moveDir) * speed * deltaTime;
123
124 if (m_SmoothCamera) {
125 // Use position directly from PerspectiveCamera
126 const glm::vec3& currentPos = perspCamera->GetPosition();
127 m_TargetPosition = currentPos + moveDir;
128 glm::vec3 newPos = currentPos;
129 SmoothDamp(newPos, m_TargetPosition, m_CurrentVelocity, m_SmoothTime, deltaTime);
130 perspCamera->SetPosition(newPos);
131 } else {
132 perspCamera->SetPosition(perspCamera->GetPosition() + moveDir);
133 }
134 }
135 }
136 }

139 void InputSystem::SmoothDamp(glm::vec3& current, const glm::vec3& target,
140 glm::vec3& velocity, float smoothTime, float deltaTime) {
141 float omega = 2.0f / smoothTime;
142 float x = omega * deltaTime;
143 float exp = 1.0f / (1.0f + x + 0.48f * x * x + 0.235f * x * x * x);
144
145 glm::vec3 change = current - target;
146 glm::vec3 temp = (velocity + omega * change) * deltaTime;
147
148 velocity = (velocity - omega * temp) * exp;
149 current = target + (change + temp) * exp;
150 }
