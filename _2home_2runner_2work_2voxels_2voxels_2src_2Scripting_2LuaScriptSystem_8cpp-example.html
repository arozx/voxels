Executes a Lua script from a provided script string.

Executes a Lua script from a provided script string.This method attempts to load and execute a Lua script directly from a string. It performs two-stage validation: first checking script loading, then script execution.

Parameters
scriptA string containing the Lua script to be executed
Returns
bool True if the script is successfully loaded and executed, false otherwise
Note
Logs detailed error messages if script loading or execution fails
Uses sol2 library for Lua script handling
Exceptions
NoneExceptions are caught and logged internally

LuaScriptSystem luaSystem; bool success = luaSystem.ExecuteScript("print('Hello, World!')"); // success will be true if script executes without errors

bool LuaScriptSystem::ExecuteScript(const std::string& script) {
sol::load_result loadResult = m_LuaState->load(script);
if (!loadResult.valid()) {
sol::error err = loadResult;
LOG_ERROR_CONCAT("Failed to load script: ", err.what());
return false;
}
sol::protected_function_result result = loadResult();
if (!result.valid()) {
sol::error err = result;
LOG_ERROR_CONCAT("Failed to execute script: ", err.what());
return false;
}
return true;
}
bool LuaScriptSystem::ExecuteFile(const std::string& originalPath) {
std::vector<std::string> searchPaths = {
originalPath, "../" + originalPath, "../../" + originalPath,
"../sandbox/assets/scripts/" + originalPath.substr(originalPath.find_last_of("/\\") + 1),
"sandbox/assets/scripts/" + originalPath.substr(originalPath.find_last_of("/\\") + 1)};
std::string validPath;
bool found = false;
for (const auto& path : searchPaths) {
if (FileSystem::Exists(path)) {
validPath = path;
found = true;
break;
}
}
if (!found) {
LOG_ERROR("Script not found: ", originalPath);
return false;
}
sol::load_result loadResult = m_LuaState->load_file(validPath);
if (!loadResult.valid() || !loadResult().valid()) {
LOG_ERROR("Failed to execute script: ", validPath);
return false;
}
return true;
}
void LuaScriptSystem::CallGlobalFunction(const std::string& functionName) {
sol::protected_function fn = (*m_LuaState)[functionName];
if (!fn.valid()) {
return;
}
sol::protected_function_result result = fn();
if (!result.valid()) {
LOG_WARN("Failed to call global function: ", functionName);
}
}
