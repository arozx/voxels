153void Profiler::EndSession() {
154 if (!m_Enabled) return;
155 if (m_ProfilingFrames) return;
156
157 switch (m_OutputFormat) {
158 case OutputFormat::JSON:
159 WriteCompleteJSON();
160 break;
161 case OutputFormat::Console:
162 LOG_INFO_CONCAT("Profile results for session", m_CurrentSession);
163 for (const auto& [name, data] : m_Profiles) {
164 if (data->empty()) continue;
165
166 LOG_INFO_CONCAT(std::string_view(name), ": Avg: ",
167 (data->totalTime / data->calls), "ms, Min: ",
168 data->minTime, "ms, Max: ", data->maxTime,
169 "ms, Calls: ", data->calls);
170 }
171 break;
172 }

196void Profiler::WriteCompleteJSON() const {
197 if (!m_Enabled) return;
198
199 using nlohmann::json;
200 json output;
201
202 std::shared_lock lock(m_Mutex);
203
204 output["session"] = m_CurrentSession;
205 output["timestamp"] = std::to_string(std::chrono::system_clock::now().time_since_epoch().count());
206 output["profiles"] = json::array();
207
208 // Add profile data first
209 for (const auto& [name, data] : m_Profiles) {
210 if (!data || data->calls == 0) continue;
211
212 json profile = {
213 {"name", std::string(std::string_view(name))},
214 {"calls", data->calls},
215 {"averageMs", data->avgTime},
216 {"recentMs", data->recentAvg},
217 {"minMs", data->minTime},
218 {"maxMs", data->maxTime},
219 {"totalMs", data->totalTime}
220 };
221
222 // Add samples array
223 profile["samples"] = json::array();
224 if (data->localCount <= data->LOCAL_SAMPLES) {
225 for (uint32_t i = 0; i < data->localCount; ++i) {
226 profile["samples"].push_back(data->localSamples[i]);
227 }
228 } else {
229 for (const auto& sample : data->samples) {
230 profile["samples"].push_back(sample);
231 }
232 }
233
234 output["profiles"].push_back(std::move(profile));
235 }
236
237 // Add frame data separately
238 if (m_ProfilingFrames || !m_FrameData.empty()) {
239 output["frame_data"] = json::array();
240 for (const auto& frame : m_FrameData) {
241 output["frame_data"].push_back({
242 {"frame_number", frame.frameNumber},
243 {"frame_time_ms", frame.frameTime},
244 {"timestamp", std::chrono::system_clock::to_time_t(frame.timestamp)}
245 });
246 }
247 }
248
249 lock.unlock();
250
251 try {
252 std::ofstream file(m_JSONOutputPath, std::ios::out | std::ios::trunc);
253 if (!file) {
254 LOG_ERROR_CONCAT("Failed to open profile output file: ", m_JSONOutputPath);
255 return;
256 }
257 file << std::setw(2) << output << std::endl;
258 file.close();
259
260 } catch (const std::exception& e) {
261 LOG_ERROR_CONCAT("Error writing profile data: ", e.what());
262 }

288void Profiler::WriteFrameDataJSON() const {
289 using nlohmann::json;
290 json output;
291
292 output["session"] = m_CurrentSession;
293 output["frames_profiled"] = m_FramesToProfile;
294 output["timestamp"] = std::to_string(std::chrono::system_clock::now().time_since_epoch().count());
295
296 // Add detailed frame data
297 output["frames"] = json::array();
298 for (const auto& frame : m_FrameData) {
299 output["frames"].push_back({
300 {"frame_number", frame.frameNumber},
301 {"frame_time_ms", frame.frameTime},
302 {"timestamp", std::chrono::system_clock::to_time_t(frame.timestamp)}
303 });
304 }
305
306 // Add frame statistics
307 if (!m_FrameData.empty()) {
308 float totalTime = 0.0f;
309 float minTime = std::numeric_limits<float>::max();
310 float maxTime = 0.0f;
311
312 for (const auto& frame : m_FrameData) {
313 totalTime += frame.frameTime;
314 minTime = std::min(minTime, frame.frameTime);
315 maxTime = std::max(maxTime, frame.frameTime);
316 }
317
318 output["statistics"] = {
319 {"total_frames", m_FrameData.size()},
320 {"average_frame_time_ms", totalTime / m_FrameData.size()},
321 {"min_frame_time_ms", minTime},
322 {"max_frame_time_ms", maxTime},
323 {"fps_average", 1000.0f / (totalTime / m_FrameData.size())},
324 {"fps_highest", 1000.0f / minTime},
325 {"fps_lowest", 1000.0f / maxTime}
326 };
327 }
328
329 try {
330 std::ofstream file(m_FramesJSONPath);
331 if (file) {
332 file << std::setw(2) << output;
333 }
334 } catch (const std::exception& e) {
335 LOG_ERROR_CONCAT("Error waiting for profile data", e.what());
336 }

354void Profiler::WriteProfile(std::string_view name, float duration) {
355 if (!m_Enabled || name.empty()) return;
356
357 // Rate limit samples to 60Hz
358 auto currentTime = std::chrono::steady_clock::now();
359 if (currentTime - s_LastSampleTime < SAMPLE_INTERVAL) {
360 return;
361 }
362 s_LastSampleTime = currentTime;
363
364 ProfileName profName(name);
365 ProfileData* data = nullptr;
366
367 {
368 std::shared_lock readLock(m_Mutex);
369 data = t_FastPath.find(profName);
370 }
371
372 if (!data) {
373 std::unique_lock lock(m_Mutex);
374 auto it = m_Profiles.find(profName);
375 if (it == m_Profiles.end()) {
376 data = m_DataPool.allocate();
377 it = m_Profiles.emplace(std::move(profName), data).first;
378 t_FastPath.insert(it->first, data);
379 } else {
380 data = it->second;
381 t_FastPath.insert(it->first, data);
382 }
383 }
384
385 data->AddSample(duration);
386 m_HasUnsavedData = true;
387
388 // Write more frequently for development
389 if ((currentTime - m_LastWriteTime) >= std::chrono::milliseconds(500)) { // Write every 500ms
390 WriteCompleteJSON();
391 m_LastWriteTime = currentTime;
392 }
393}

440bool Profiler::EndFrame() {
441 if (!m_ProfilingFrames) return false;
442
443 // Calculate frame time
444 static auto lastFrameTime = std::chrono::steady_clock::now();
445 auto currentTime = std::chrono::steady_clock::now();
446 float frameTime = std::chrono::duration<float, std::milli>(currentTime - lastFrameTime).count();
447 lastFrameTime = currentTime;
448
449 // Record frame data
450 m_FrameData.emplace_back(m_CurrentProfiledFrame, frameTime);
451
452 m_CurrentProfiledFrame++;
453
454 if (m_CurrentProfiledFrame >= m_FramesToProfile) {
455 m_ProfilingFrames = false;
456 WriteFrameDataJSON(); // Write frame data to separate file
457 m_FrameData.clear();
458
459 RestoreProfilingState(); // Restore previous profiling state
460 LOG_INFO("Frame profiling complete, resumed normal profiling.");
461 return false;
462 }
463 return true;
464}

478void Profiler::RestoreProfilingState() {
479 // Clean up current frame profiling data
480 for (auto& [name, data] : m_Profiles) {
481 m_DataPool.deallocate(data);
482 }
483 m_Profiles.clear();
484 t_FastPath = FastPathCache(); // Reset fast path cache
485
486 // Restore the original profiles
487 for (const auto& [name, data] : m_RegularProfiles) {
488 auto* newData = m_DataPool.allocate();
489 *newData = *data; // Copy the data back
490 m_Profiles[name] = newData;
491 t_FastPath.insert(name, newData); // Re-populate fast path cache
492 }
493
494 // Clean up temporary storage
495 for (auto& [name, data] : m_RegularProfiles) {
496 m_DataPool.deallocate(data);
497 }
498 m_RegularProfiles.clear();
499
500 // Restore original session name and force a write
501 m_CurrentSession = m_RegularSession; // Directly restore session name
502 m_HasUnsavedData = true; // Force a write of restored data
503 WriteCompleteJSON(); // Write the restored data immediately
504
505 // Reset frame-specific state
506 m_ProfilingFrames = false;
507 m_CurrentProfiledFrame = 0;
508 m_FrameData.clear();
509 m_BatchedMeasurements.clear();
510
511 // Restore fast path cache and enable profiling
512 m_Enabled = true;
513 t_FastPath = FastPathCache(); // Reset cache
514
515 // Restore session and force a write
516 BeginSession(m_RegularSession); // Use BeginSession instead of direct assignment
517
518 // Re-register all existing profiles in the fast path cache
519 for (const auto& [name, data] : m_Profiles) {
520 t_FastPath.insert(name, data);
521 }
522
523 m_HasUnsavedData = true;
524 m_LastWriteTime = std::chrono::steady_clock::now();
525 WriteCompleteJSON();
526}
