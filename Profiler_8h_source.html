313 struct alignas(64) ProfileData {
314 static constexpr size_t LOCAL_SAMPLES = 8;
315 static constexpr size_t MAX_SAMPLES = 1000; // Limit total samples
316
317 float localSamples[LOCAL_SAMPLES];
318 SmallVector<float, SMALL_VECTOR_SIZE> samples;
319 uint32_t calls{0};
320 uint32_t localCount{0};
321 float minTime{std::numeric_limits<float>::max()};
322 float maxTime{std::numeric_limits<float>::lowest()};
323 float totalTime{0.0f};
324 float avgTime{0.0f};
325 float recentAvg{0.0f}; // Moving average for recent samples
326
327 void AddSample(float time) noexcept {
328 calls++;
329
330 // Update moving average
331 constexpr float alpha = 0.1f; // Smoothing factor
332 recentAvg = alpha * time + (1.0f - alpha) * recentAvg;
333
334 if (localCount < LOCAL_SAMPLES) {
335 localSamples[localCount++] = time;
336 } else {
337 if (localCount == LOCAL_SAMPLES) {
338 samples.moveToHeap();
339 for (uint32_t i = 0; i < LOCAL_SAMPLES; i++) {
340 samples.push_back(localSamples[i]);
341 }
342 localCount++;
343 }
344
345 // Implement circular buffer behavior when max samples reached
346 if (samples.size() >= MAX_SAMPLES) {
347 // Shift samples left, discarding oldest
348 std::rotate(samples.begin(), samples.begin() + 1, samples.end());
349 samples.set_back(time);
350 } else {
351 samples.push_back(time);
352 }
353 }
354
355 minTime = std::min(minTime, time);
356 maxTime = std::max(maxTime, time);
357 totalTime += time;
358 avgTime = totalTime / static_cast<float>(calls);
359 }
360
361 bool empty() const { return samples.empty(); }
362 size_t size() const { return samples.size(); }
363 auto begin() const { return samples.begin(); }
364 auto end() const { return samples.end(); }
365
366 // Add copy constructor and assignment operator
367 ProfileData& operator=(const ProfileData& other) {
368 if (this != &other) {
369 samples = other.samples;
370 calls = other.calls;
371 localCount = other.localCount;
372 minTime = other.minTime;
373 maxTime = other.maxTime;
374 totalTime = other.totalTime;
375 avgTime = other.avgTime;
376 recentAvg = other.recentAvg;
377 std::memcpy(localSamples, other.localSamples, sizeof(localSamples));
378 }
379 return *this;
380 }

484 std::array<Entry, FAST_PATH_SIZE> entries;

487 ProfileData* find(const ProfileName& name) {
488 accessCount++;
489 for (auto& entry : entries) {

492 return entry.data;
493 }

495 return nullptr;

498 void insert(const ProfileName& name, ProfileData* data) {
499 accessCount++;
500 // Find least recently used entry
501 size_t lru_idx = 0;
502 uint64_t oldest = UINT64_MAX;
503 for (size_t i = 0; i < FAST_PATH_SIZE; i++) {
504 if (!entries[i].data) {
505 lru_idx = i;
506 break;
507 }
508 if (entries[i].lastAccess < oldest) {
509 oldest = entries[i].lastAccess;
510 lru_idx = i;
511 }
512 }
513 // Ensure lru_idx is within bounds
514 if (lru_idx < FAST_PATH_SIZE) {
515 entries[lru_idx] = {name, data, accessCount};

517 }
