163void Renderer2D::DrawQuad(const glm::vec3& position, const glm::vec2& size,
164 const glm::vec4& color) {
165 if (s_Data.QuadIndexCount >= Renderer2DData::MaxIndices) NextBatch();
166
167 const float texIndex = 0.0f; // White Texture
168 const float tilingFactor = 1.0f;
169
170 glm::mat4 transform = glm::translate(glm::mat4(1.0f), position) *
171 glm::scale(glm::mat4(1.0f), {size.x, size.y, 1.0f});
172
173 // Add vertices to buffer
174 for (size_t i = 0; i < 4; i++) {
175 s_Data.QuadVertexBufferPtr->Position = transform * s_Data.QuadVertexPositions[i];
176 s_Data.QuadVertexBufferPtr->Color = color;
177 s_Data.QuadVertexBufferPtr->TexCoord = {(i == 1 || i == 2) ? 1.0f : 0.0f,
178 (i == 2 || i == 3) ? 1.0f : 0.0f};
179 s_Data.QuadVertexBufferPtr->TexIndex = texIndex;
180 s_Data.QuadVertexBufferPtr->TilingFactor = tilingFactor;
181 s_Data.QuadVertexBufferPtr++;
182 }

185}
